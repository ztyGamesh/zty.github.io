随机生成指定长度的字符串

function randomStr(n) {
  let starndard = 'abcdefghijklmnopqrstuvwxyz9876543210';
  let len = starndard.length;
  let result = '';

  for (let i = 0; i < n; i++) {
    result += starndard.charAt(Math.floor(Math.random() * len))
  }

  return result
}
